Example

<form name='form' w-form-for='samurai' w-submit='send()'>
  <div w-field='name'
       type='w-combo'
       items='names'
       label='Name'
       ng-required='true'
       ng-minlength='2'
       ng-maxlength='4'
       ng-pattern='/J/'></div>

  <div w-field='age'
       type='w-chz'
       items='ages'
       label='Age'
       ng-disabled='true'></div>

  <div w-field='gender'
       type='w-radio'
       items='genders'
       label='Gender'></div>

  <div w-field='likesToEat'
       type='w-checkbox'
       items='meals'
       label='Likes to eat'></div>

  <div w-field='foes'
       type='w-multi-select'
       items='warriors'
       label='Foes'></div>

  <div w-field='aspirations'
       type='w-tags'
       items='goals'
       label='Aspirations'></div>

  <div w-submit-field>Save</div>
</form>
        
angular.module('angular-w-demo')
    .controller('FormBuilderCtrl', function($scope, $http) {
  $scope.samurai = {};
  $scope.stages = [
    {id: 'S', label: 'Shu'},
    {id: 'H', label: 'Ha'},
    {id: 'R', label: 'Ri'}
  ];
  $scope.names = ["Joe", "Sue", "Sam"];
  $scope.ages = []
  for (var age = 0; age < 100; age++) {
    $scope.ages.push({
      id: age,
      label: age
    });
  }

  $scope.meals = [
    {id: 'meat', label: 'meat'},
    {id: 'milk', label: 'milk'},
    {id: 'orange', label: 'orange'}
  ];
  $scope.weapons = [
    {id: '1', label: 'Katana'},
    {id: '2', label: 'Naginata'},
    {id: '3', label: 'Yari'}
  ];
  $scope.weaponStyles = ['Kenjutsu', 'Naginatajutsu', 'Sōjutsu']

  $scope.send = function() {
    var res = $http.post('/').success(function(data) {
      angular.copy(data, $scope.samurai);
    });
  }
});
