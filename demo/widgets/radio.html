Radio

Description

Directive: w-radio

Provides default radio-like behaviour. Supports several angular directives: ng-model, ng-disabled, ng-required. Available option is items.

Example

<form role="form" ng-controller='RadioCtrl'>
  <div class="form-group">
    <label>Choose your katawari</label>
    <div w-radio="true"
         items="items"
         ng-model="value"></div>
  </div>
</form>
angular.module('angular-w-demo')
.controller('RadioCtrl', function($scope) {
  $scope.items = [
    {id: 'S', label: 'Shijima'},
    {id: 'M', label: 'Musubi'},
    {id: 'Y', label: 'Yosuga'}
  ];
});
