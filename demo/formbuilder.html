Form Builder

Description

Directive: fs-form-for

Form Builder creates a form for the provided object using declarative style. Input fields are created by fs-field directive.

Example

<form name='form'
    fs-form-for='samurai'
    fs-submit='send()'>

<div fs-field='name'
     type='fs-combo'
     items='names'
     label='Name'></div>

<div fs-field='age'
     type='fs-chz'
     items='ages'
     label='Age'></div>

<div fs-field='stage'
     type='fs-radio'
     items='stages'
     label='Learning Stage'></div>

<div fs-field='likesToEat'
     type='fs-checkbox'
     items='meals'
     label='Likes to eat'></div>

<div fs-field='weapons'
     type='fs-multi-select'
     items='weapons'
     label='Weapons'></div>

<div fs-field='styles'
     type='fs-tags'
     items='weaponStyles'
     label='Styles'></div>

<div fs-submit-field>Save</div>
</form>
      
angular.module('formstamp-demo')
  .controller('FormBuilderCtrl', function($scope, $http) {
$scope.samurai = {};
$scope.stages = [
  {id: 'S', label: 'Shu'},
  {id: 'H', label: 'Ha'},
  {id: 'R', label: 'Ri'}
];
$scope.names = ["Joe", "Sue", "Sam"];
$scope.ages = []
for (var age = 0; age < 100; age++) {
  $scope.ages.push({
    id: age,
    label: age
  });
}

$scope.meals = [
  {id: 'meat', label: 'meat'},
  {id: 'milk', label: 'milk'},
  {id: 'orange', label: 'orange'}
];
$scope.weapons = [
  {id: '1', label: 'Katana'},
  {id: '2', label: 'Naginata'},
  {id: '3', label: 'Yari'}
];
$scope.weaponStyles = ['Kenjutsu', 'Naginatajutsu', 'Sōjutsu']

$scope.send = function() {
  var res = $http.post('/').success(function(data) {
    angular.copy(data, $scope.samurai);
  });
}
});

Directive: fs-field

        

          fs-field creates an input for the specified attribute of the
          object passed to fs-form-for and displays validation errors.
        

      

    

  


  

    

      
Option type

      Required: true
      

        Specifies input type. Allowed values:
        fs-chz, fs-multiselect, fs-combo,
        fs-multiselect, fs-radio, fs-checkbox.
      

    

  


  

    

      
Option items

      Required: true
      

        Specifies items passed to input.
      

    

  


  

    

      
Option label

      Required: false
      

        Specifies label.
      

    

  


  

    

      
Option ng-disabled

      Required: false
      

        Disables input.
      

      
Example

    

  

  

    

      
<form name='form' fs-form-for='samurai' fs-submit='send()'>
<div fs-field='province'
     type='fs-chz'
     items='provinces'
     label='Province'
     ng-disabled='true'></div>
</form>
      

    

    

      

        

      

    

  


  

    

      
Validation options:

      

        ng-required - sets required attribute if set to true.
        Supported by all types of input.
      

      

        ng-minlength - sets minlength validation error key if the value is shorter than minlength.
        Supported by combo and tags.
      

      

        ng-maxlength - sets maxlength validation error key if the value is longer than maxlength.
        Supported by combo and tags.
      

      

        ng-pattern - Sets pattern validation error key if the value does not match the RegExp pattern expression.
        Expected value is /regexp/ for inline patterns or regexp for patterns defined as scope expressions.
        Supported by combo and tags.
      

    

  

  

    

      
Example

      
<form name='form' fs-form-for='ronin' fs-submit='send()'>
<div fs-field='name'
     type='fs-combo'
     items='names'
     label='Name'
     ng-pattern='/J/'></div>
<div fs-field='age'
     type='fs-chz'
     items='ages'
     label='Age'
     ng-required='true'></div>
<div fs-field='style'
     type='fs-combo'
     items='weaponStyles'
     label='Style'
     ng-minlength='3'
     ng-maxlength='10'></div>
</form>
      

    

    

      

        

        

        

      

    

  


  

    

      

        
Directive: fs-submit-field

        

          fs-submit-field creates a form submit button. The form will not be submitted if it is invalid.
